[docs]    def history(self, ticker_code: str, start: datetime, end: datetime) -> pd.DataFrame:
        kbars = self.api.kbars(
            Stock(self.api, ticker_code),
            start=start,
            end=end
        )
        df = pd.DataFrame({**kbars})
        df.ts = pd.to_datetime(df.ts)
        return df

[docs]    def trade(self,
              ticker_code: str,
              price: float,
              quantity: int = 1,
              action: Action = Action.Buy,
              price_type: StockPriceType = StockPriceType.LMT,
              order_type: StockOrderType = StockOrderType.Common,
              order_lot: str = STOCK_ORDER_LOT_COMMON
              ) -> Trade:
        order = self.api.Order(
            price=price,
            quantity=quantity,
            action=action,
            price_type=price_type,
            order_type=order_type,
            order_lot=order_lot,
            account=self.api.stock_account
        )
        return self.api.place_order(
            Stock(self.api, ticker_code),
            order
        )
